import socket

HOST = "0.0.0.0"
PORT = 8002   # client 2 écoute ici

s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.bind((HOST, PORT))
s.listen(5)

print(" Client 2 en écoute sur le port 8002")

while True:
    conn, addr = s.accept()
    data = conn.recv(4096).decode()
    print(f" Message reçu : {data}")

    # réponse simple
    response = f"Réponse de Client 2 à : {data}"
    conn.send(response.encode())
    conn.close()
